Étapes de déploiement
1. Créez un compte gratuit sur supabase.com
2. Nouveau projet → Settings → API → copiez URL + anon key
3. SQL Editor → exécutez le script ci-dessous
4. Collez les clés ci-dessus et cliquez Connecter
5. Uploadez index.html sur GitHub → activez GitHub Pages ou connectez Vercel
Script SQL — à coller dans Supabase SQL Editor
create table if not exists journees (
  id uuid primary key default gen_random_uuid(),
  restaurant text not null default 'SAINT CLEMENT',
  date date not null unique,
  -- CA
  ca_ht numeric, ca_ttc numeric,
  ca_n1 numeric, ca_prevu numeric,
  -- Tickets
  tickets_nb int, tickets_n1 int,
  ticket_moyen_ttc numeric,
  -- Kiosk
  kiosk_ttc numeric, kiosk_tickets int,
  kiosk_place_ttc numeric, kiosk_emporter_ttc numeric,
  -- Drive
  drive_ttc numeric, drive_voitures int,
  drive_n1 numeric, drive_ticket_moyen numeric,
  -- Livraison
  livraison_ttc numeric, livraison_commandes int,
  livraison_n1 numeric, livraison_ticket_moyen numeric,
  -- Comptoir
  comptoir_ttc numeric, comptoir_tickets int,
  -- Click & Collect
  cc_ttc numeric, cc_tickets int,
  -- Mode conso
  sur_place_tickets int, emporter_tickets int,
  -- Encaissements
  cb numeric, especes numeric,
  uber_eats numeric, deliveroo numeric,
  bk_livraison numeric, trd numeric, ancv numeric,
  erreur_caisse numeric,
  -- Corrections
  corrections_nb int, corrections_montant numeric, corrections_pct numeric,
  -- Remises
  remises_nb int, remises_montant numeric, remises_pct numeric,
  -- Sauces offertes
  sauces_nb int, sauces_montant numeric, sauces_pct numeric,
  -- Annulations
  annulations_nb int, annulations_montant numeric,
  -- Commandes ouvertes
  commandes_ouvertes_nb int, commandes_ouvertes_montant numeric,
  -- Repas employés
  repas_employes_nb int, repas_employes_montant numeric, repas_employes_pct numeric,
  -- Ventes annexes
  ventes_annexes_ttc numeric, ventes_annexes_nb int,
  -- MO & RH (export séparé)
  mo_pct numeric, salaires numeric,
  -- Temps de service (export séparé)
  ts_bornes_midi numeric, ts_drive_midi int,
  ts_bornes_soir numeric, ts_drive_soir int,
  -- Pertes (export séparé)
  pertes_brutes numeric, pertes_finies numeric,
  -- Commentaire
  commentaire text,
  created_at timestamptz default now()
);
-- RLS
alter table journees enable row level security;
create policy "lecture publique" on journees for select using (true);
create policy "ecriture" on journees for insert with check (true);
create policy "mise a jour" on journees for update using (true);
